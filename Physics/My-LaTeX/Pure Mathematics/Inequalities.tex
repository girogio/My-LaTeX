	\section{Quadratic Inequalities}
	\begin{example}
		Solve the inequality $x^2-2x>3$
	\end{example}
	
	\begin{center}
		$$x^2-2x>3$$
		$$\implies x^2-2x-3>0$$
		$$\text{Let }x^2-2x-3=0$$
		$$\implies(x-3)(x+1)=0$$
	\end{center}	
	\begin{center}
		\begin{tikzpicture}

	\end{center}
